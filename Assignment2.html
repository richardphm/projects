create_linegraph <- function(x) {
  ggplot(data = student_debt) +
  filter(race == "White") %>%
  aes(x = x, y = loan_debt_pct) +
  geom_line() +
  labs(x = "Year", y = "Loan Debt Percentage", caption = "Student loan debt percentage for white Americans over time.", title = "Student Loan Debt Percentage for White Americans") +
  theme(plot.caption = element_text(hjust = 0))
}
create_linegraph(student_debt$year)
